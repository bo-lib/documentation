  drillItems: Array<any> = []; // Itens da tabela.

  constructor(private changeDetector: ChangeDetectorRef) {}

  /**
   * Lifecycle hook que é chamado quando qualquer propriedade vinculada a dados de uma diretiva é alterada.
   * @param changes - mudanças.
   * @autor Matheus Camargo B. da Costa
   */
  ngOnChanges(changes: SimpleChanges): void {
    if (changes['drillDownData']) {
      const previousValue = changes['drillDownData'].previousValue;
      const currentValue = changes['drillDownData'].currentValue;
      this.pagination = this.paginationType == 'loadMore' ? false : true;
      if (JSON.stringify(previousValue) !== JSON.stringify(currentValue)) {
        this.updateComponent();
      } else {
        this.drillIsLoading = false;
      }
    }

    if (changes['allDrillDownData']) {
      const previousValue = changes['allDrillDownData'].previousValue;
      const currentValue = changes['allDrillDownData'].currentValue;

      if (JSON.stringify(previousValue) !== JSON.stringify(currentValue)) {
        this.drillIsLoading = false;
      } else {
        this.drillIsLoading = false;
      }
    }

    if (!this.drillInitiated) {
      this.addBreadcrumb();
      this.drillInitiated = true;
    }
  }

  /**
   * Realiza a paginação com base nos dados fornecidos via Input.
   * Atualiza os itens paginados, página atual, itens por página e total de páginas.
   * @autor Matheus Camargo B. da Costa
   */
  paginationByInput(): void {
    if(!this.pagination) return;

    this.paginatedItems = this.drillDownData.items;
    this.currentPage = this.paginationInput.currentPage;
    this.itemsPerPage = this.paginationInput.itemsPerPage;
    this.totalPages = this.paginationInput.totalPages;
  }

  /**
   * Realiza a paginação com base nos dados do array local.
   * Calcula o total de itens e chama o método para paginar os itens.
   * @autor Matheus Camargo B. da Costa
   */
  paginationByArray(): void {
    if(!this.pagination) return;

    this.totalItems = this.drillDownData.items.length;
    this.paginateItems();
  }

  /**
   * Emite o evento para navegar para a próxima página.
   * @autor Matheus Camargo B. da Costa
   */
  nextPageEvent(): void {
    if(!this.pagination) return;

    this.drillIsLoading = true;
    this.paginateEventClicked.emit({ type: 'next_page' });
  }

  /**
   * Emite o evento para navegar para a página anterior.
   * @autor Matheus Camargo B. da Costa
   */
  previousPageEvent(): void {
    if(!this.pagination) return;

    this.drillIsLoading = true;
    this.paginateEventClicked.emit({ type: 'previous_page' });
  }

  /**
   * Emite o evento para navegar para a primeira página.
   * @autor Matheus Camargo B. da Costa

  firstPageEvent(): void {
    if(!this.pagination) return;

    this.drillIsLoading = true;
    this.paginateEventClicked.emit({ type: 'first_page' });
  }

  /**
   * Emite o evento para navegar para a última página.

   */
  lastPageEvent(): void {
    if(!this.pagination) return;

    this.drillIsLoading = true;
    this.paginateEventClicked.emit({ type: 'last_page' });
  }

   * @param quantity - Quantidade de itens por página.

  changeQuantityPerPageEvent(quantity: number): void {
    if(!this.pagination) return;
    this.drillIsLoading = true;
    this.paginateEventClicked.emit({
      type: 'change_quantity_per_page',
      value: quantity,
    });

  /**
   * Atualiza o componente com novos dados.
   * @autor Matheus Camargo B. da Costa
   */
  updateComponent(): void {
    if (
      this.drillDownData.header.length === 0 ||
      this.drillDownData.items.length === 0
    )
      return;
    this.drillColumns = this.drillDownData.header.map((column) => {
      return column.type === 'link'
        ? { ...column, type: 'cellTemplate' }
        : {
            ...column,
          };
    });
    this.drillItems = this.drillDownData.items;
    this.drillIsLoading = false;

    if (!this.isPaginationByInput) this.paginationByArray();
    else this.paginationByInput();

    if (this.hasActions) {
      for (let action of this.actionsData) {
        this.actions.push({
          icon: action.icon,
          label: action.label,
          action: () => {
            this.actionEvent(action.key);
          },
        });
      }
    } else {
      this.actions = [];
    }
  }

  /**
   * Navega um nivel para baixo na hierarquia de drilldown.
   * @param row - Row selecionada.
   * @param clickedValue - Dado do link que clicou.
   * @autor Matheus Camargo B. da Costa
   */
  navigateDown(row: any, clickedValue: string): void {
    if (!row || !clickedValue) {
      console.error('Erro: Propriedade inexistente ou valor inválido');
      return;
    }
    this.addBreadcrumb(row, clickedValue);
    this.drillIsLoading = true;
    this.hyperlinkClicked.emit({ row, clickedValue, action: 'Link' });
  }

  /**
   * Navega um nivel para cima na hierarquia de drilldown.
   * @param row - Row selecionada.
   * @param clickedValue - Dado do link que clicou.
   * @autor Matheus Camargo B. da Costa
   */
  navigateUp(row: any, clickedValue: any): void {
    this.getBackToBreadcrumb(clickedValue);
    this.drillIsLoading = true;
    this.hyperlinkClicked.emit({ row, clickedValue, action: 'Link' });
  }

  /**
   * Carrega mais itens da lista.
   * @autor Matheus Camargo B. da Costa
   */
  loadMoreItems(): void {
    this.drillIsLoading = true;
    this.loadMoreClicked.emit();
  }

  /**
   * Adiciona um breadcrumb para a bandeja de breadcrumbs.
   * @param row - Row selecionada (opcional).
   * @param clickedValue - Valor clicado (opcional).
   * @autor Matheus Camargo B. da Costa
   */
  addBreadcrumb(row: any = null, clickedValue: string = ''): void {
    const breadcrumbItem: PoBreadcrumbItem = {
      label: row == null ? this.drillDownName : clickedValue,
      action: () => {
        if (row) this.navigateUp(row, clickedValue);
        else this.getBackToRoot(this.drillDownName);
      },
    };

    for (let breacrumb of this.drillBreadcrumbs) {
      if (breacrumb.label === breadcrumbItem.label) {
        return;
      }
    }

    this.drillBreadcrumbs.push(breadcrumbItem);
    this.changeDetector.detectChanges();
  }

  /**
   * Navega de volta para uma breadcrumb especifica.
   * @param clickedValue - Valor clicado.
   * @autor Matheus Camargo B. da Costa
   */
  getBackToBreadcrumb(clickedValue: any): void {
    const index = this.drillBreadcrumbs.findIndex(
      (breadcrumb) => breadcrumb.label === clickedValue
    );
    if (index !== -1) {
      this.drillBreadcrumbs = this.drillBreadcrumbs.slice(0, index + 1);
    }
  }

  /**
   * Navega de volta para raiz da tabela.
   * @param clickedValue - Valor clicado.
   * @autor Matheus Camargo B. da Costa
   */
  getBackToRoot(clickedValue: any): void {
    const index = this.drillBreadcrumbs.findIndex(
      (breadcrumb) => breadcrumb.label === clickedValue
    );
    if (index !== -1) {
      this.drillBreadcrumbs = this.drillBreadcrumbs.slice(0, index + 1);
    }
    this.drillIsLoading = true;
    this.hyperlinkClicked.emit(null);
  }

  getAllItems(): void {
    this.getAllItemsToExport.emit();
  }

  /**
   * Retorna dados selecionados.
   * @autor Matheus Camargo B. da Costa
   */
  selectedEvent(event: any): void {
    this.selectedItems.push(event);
    this.getSelectableEvent.emit(this.selectedItems);
  }

  /**
   * Retorna evento desmarcado.
   * @autor Matheus Camargo B. da Costa
   */
  unselectedEvent(event: any): void {
    this.selectedItems = this.selectedItems.filter((item) => item !== event);
    this.getSelectableEvent.emit(this.selectedItems);
  }

  /**
   * Retorna todos os itens selecionados.
   * @autor Matheus Camargo B. da Costa
   */
  allSelectedEvent(event: any[]): void {
    this.selectedItems = event;
    this.getSelectableEvent.emit(this.selectedItems);
  }

  /**
   * Retorna todos os itens desmarcados.
   * @autor Matheus Camargo B. da Costa
   */
  allUnselectedEvent(event: any[]): void {
    this.selectedItems = [];
    this.getSelectableEvent.emit(this.selectedItems);
  }

  /**
   * Evento de ação.
   * @param actionType - Tipo da ação.
   * @autor Matheus Camargo B. da Costa
   */
  actionEvent(actionType: any): void {
    this.getActionEvent.emit(actionType);
  }

  /**
   * Realiza a paginação dos itens com base na página atual e itens por página.
   * @autor Matheus Camargo B. da Costa
   */
  paginateItems(): void {
    if(!this.pagination) return;
    const startIndex = (this.currentPage - 1) * this.itemsPerPage;
    const endIndex = startIndex + this.itemsPerPage;
    this.paginatedItems = this.drillDownData.items.slice(startIndex, endIndex);
    console.log(this.paginatedItems);
  }

  /**
   * Navega para uma página específica.
   * @param page - Número da página para navegar.
   * @autor Matheus Camargo B. da Costa
   */
  goToPage(page: number): void {
    if(!this.pagination) return;
    if (page < 1 || page > this.getTotalPages()) return;
    this.currentPage = page;
    this.paginateItems();
  }

  /**
   * Retorna o total de páginas com base no total de itens e itens por página.
   * @returns Número total de páginas.
   * @autor Matheus Camargo B. da Costa
   */
  getTotalPages(): number {
    return Math.ceil(this.totalItems / this.itemsPerPage);
  }

  /**
   * Navega para a próxima página.
   * @autor Matheus Camargo B. da Costa
   */
  nextPage(): void {
    if(!this.pagination) return;

    if (this.currentPage < this.getTotalPages()) {
      this.currentPage++;
      this.paginateItems();
    }
  }

  /**
   * Navega para a página anterior.
   * @autor Matheus Camargo B. da Costa
   */
  previousPage(): void {
    if(!this.pagination) return;

    if (this.currentPage > 1) {
      this.currentPage--;
      this.paginateItems();
    }
  }

  /**
   * Navega para a primeira página.
   * @autor Matheus Camargo B. da Costa
   */
  firstPage(): void {
    if(!this.pagination) return;

    this.currentPage = 1;
    this.paginateItems();
  }

  /**
   * Navega para a última página.
   * @autor Matheus Camargo B. da Costa
   */
  lastPage(): void {
    if(!this.pagination) return;

    this.currentPage = this.getTotalPages();
    this.paginateItems();
  }

  /**
   * Atualiza a quantidade de itens por página e reinicia a paginação.
   * @param newItemsPerPage - Nova quantidade de itens por página.
   * @autor Matheus Camargo B. da Costa
   */
  onItemsPerPageChange(newItemsPerPage: number): void {
    if(!this.pagination) return;

    this.itemsPerPage = newItemsPerPage;
    this.currentPage = 1;
    if (this.isPaginationByInput)
      this.changeQuantityPerPageEvent(newItemsPerPage);
    this.paginateItems();
  }
}
